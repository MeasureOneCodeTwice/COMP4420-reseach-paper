% Math blocks:
% - `align` (without the asterisk) will place numbers for each line

\begin{enumerate}
    \item Lists can be bullets, numbered, or other forms of markers
    \begin{enumerate}
        \item can also nest the list
    \end{enumerate}
    \item Math inline: $O(n) = \# \ \textbf{special symbols must be escaped}$
    \begin{enumerate}
        \item Can add a space by simply escaping it via \verb|\ |
    \end{enumerate}
    \item Math blocks can be wrapped with \verb|\[ ... \]| or \verb|begin{math}|
    \begin{enumerate}
        \item Aligning multiple math lines by the same symbol: use the \verb|align| argument and then prepend \verb|&| to the symbol you want to align and add \verb|\\| where you want to break the line
        \item FYI, can also do a code block with \verb|\begin{verbatim}|
    \end{enumerate}
    \item \verb|log| and some other math terms work best when escaped, so \verb|\log| and \verb|\bmod| vs just \verb|log| and \verb|mod|
    \item If you want parentheticals, you can use \verb|()|, \verb|{}|, \verb|[]|, or anything else I guess. However, sometimes you want them to be \emph{smart} parentheticals so that they resize to fit the context: $(\frac{N^M}{B})$ vs $\left( \frac{N^M}{B} \right)$ or just manually make them big with $\Bigg( \frac{N}{B} \Bigg)$. \href{https://www.overleaf.com/learn/latex/Brackets_and_Parentheses}{More information here.}
\end{enumerate}
